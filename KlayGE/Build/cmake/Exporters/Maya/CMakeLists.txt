REMOVE_DEFINITIONS(-DUNICODE -D_UNICODE)

SET(KLAYGE_MAYA_PLUGIN_DIR "${KLAYGE_MAYA_PATH}/bin/plug-ins")

SET(PLUGIN_NAME "MayaMeshMLPlugin")

SET(SOURCE_FILES
	${KLAYGE_PROJECT_DIR}/Exporters/src/Maya/exporter.cpp
	${KLAYGE_PROJECT_DIR}/Exporters/src/Maya/export.def
)

SOURCE_GROUP("Source Files" FILES ${SOURCE_FILES})

ADD_DEFINITIONS(-D_BOOL)
ADD_DEFINITIONS(-DNT_APP)
ADD_DEFINITIONS(-DREQUIRE_IOSTREAM)

INCLUDE_DIRECTORIES(${KLAYGE_MAYA_PATH}/include)
INCLUDE_DIRECTORIES(${Boost_INCLUDE_DIR})
INCLUDE_DIRECTORIES(${KLAYGE_PROJECT_DIR}/Core/Include)
INCLUDE_DIRECTORIES(${KLAYGE_PROJECT_DIR}/../KFL/include)
INCLUDE_DIRECTORIES(${KLAYGE_PROJECT_DIR}/../MeshMLLib/include)
LINK_DIRECTORIES(${KLAYGE_MAYA_PATH}/lib)
IF(MSVC)
	IF(MSVC_VERSION LESS 1699)
		LINK_DIRECTORIES(${Boost_LIBRARY_DIR})
	ENDIF()
ELSE()
	LINK_DIRECTORIES(${Boost_LIBRARY_DIR})
ENDIF()
LINK_DIRECTORIES(${KLAYGE_PROJECT_DIR}/../KFL/lib/${KLAYGE_PLATFORM_NAME})
LINK_DIRECTORIES(${KLAYGE_PROJECT_DIR}/../MeshMLLib/lib/${KLAYGE_PLATFORM_NAME})

ADD_LIBRARY(${PLUGIN_NAME} SHARED ${SOURCE_FILES} ${HEADER_FILES} ${RESOURCE_FILES})

SET_TARGET_PROPERTIES(${PLUGIN_NAME} PROPERTIES
	PROJECT_LABEL ${PLUGIN_NAME}
	DEBUG_POSTFIX ""
	OUTPUT_NAME "MeshML"
)

SET(EXTRA_LINKED_LIBRARIES Foundation OpenMaya OpenMayaAnim)
ADD_DEPENDENCIES(${PLUGIN_NAME} ${KLAYGE_CORELIB_NAME})

TARGET_LINK_LIBRARIES(${PLUGIN_NAME}
	debug MeshMLLib${KLAYGE_OUTPUT_SUFFIX}${CMAKE_DEBUG_POSTFIX} optimized MeshMLLib${KLAYGE_OUTPUT_SUFFIX}
	${EXTRA_LINKED_LIBRARIES})


ADD_CUSTOM_COMMAND(TARGET ${PLUGIN_NAME}
	POST_BUILD
	COMMAND ${CMAKE_COMMAND} -E copy_if_different $<TARGET_FILE_DIR:${PLUGIN_NAME}>/$<TARGET_FILE_NAME:${PLUGIN_NAME}>
		${KLAYGE_MAYA_PLUGIN_DIR}/MeshML.mll)


INSTALL(TARGETS ${PLUGIN_NAME}
	RUNTIME DESTINATION ${KLAYGE_MAYA_PLUGIN_DIR}
	LIBRARY DESTINATION ${KLAYGE_MAYA_PLUGIN_DIR}
)

CREATE_VCPROJ_USERFILE(${PLUGIN_NAME})

SET_TARGET_PROPERTIES(${PLUGIN_NAME} PROPERTIES FOLDER "Tools/Exporters")
